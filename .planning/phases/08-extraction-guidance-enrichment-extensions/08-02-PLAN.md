```python
# --- DropTarget classification (D-05, D-06, D-07) ---

_BIDDER_WITHDRAWAL_SIGNALS = (
    "could not improve",
    "would not improve",
    "would not be in a position",
    "no longer interested",
    "unable to proceed",
    "would not move forward",
    "not prepared to move forward",
    "declined to improve",
    "was not able to increase",
    "not in a position to actively conduct",
    "could not increase",
    "would not continue",
    "cannot proceed",
    "would not proceed",
)


def _is_bidder_signaled_withdrawal(drop_reason_text: str | None) -> bool:
    """D-07: Check if bidder signaled withdrawal first."""
    if not drop_reason_text:
        return False
    lower = drop_reason_text.lower()
    return any(signal in lower for signal in _BIDDER_WITHDRAWAL_SIGNALS)


def _classify_dropouts(
    events: list[SkillEventRecord],
    rounds: list[dict],
) -> dict[str, dict]:
    """Deterministic DropTarget classification per D-05/D-06/D-07.

    Only emits DropTarget when target-initiated exclusion is filing-grounded
    by round invitation context. Bidder-signaled withdrawals are excluded
    even when the target later confirms exclusion.
    """
    result: dict[str, dict] = {}
    event_order = [e.event_id for e in events]

    # Build set of actor_ids invited to each round
    # Map: round announcement index -> set of invited actor_ids
    round_invited: list[tuple[int, set[str]]] = []
    for r in rounds:
        ann_id = r["announcement_event_id"]
        if ann_id not in event_order:
            continue
        ann_idx = event_order.index(ann_id)
        invited = set(r.get("invited_actor_ids") or [])
        if invited:  # Only consider rounds with explicit invitations
            round_invited.append((ann_idx, invited))

    # Track which bidders are active (have NDA, no prior drop)
    # within restart-delimited cycles
    cycle_ranges = _cycle_ranges(events)
    bidder_ids_all = {
        aid for evt in events for aid in evt.actor_ids
        if evt.event_type in ("nda", "proposal", "drop")
    }

    for evt_idx, evt in enumerate(events):
        if evt.event_type != "drop":
            continue

        # D-07: Check bidder withdrawal signal FIRST
        if _is_bidder_signaled_withdrawal(evt.drop_reason_text):
            continue

        drop_actor_ids = set(evt.actor_ids)
        if not drop_actor_ids:
            continue

        # Check if any dropped actor was active but not invited to a
        # round announced before this drop event
        excluded = False
        for ann_idx, invited in round_invited:
            if ann_idx >= evt_idx:
                continue  # Round announced after drop -- not relevant
            for actor_id in drop_actor_ids:
                if actor_id not in invited:
                    # Actor was NOT invited to this round
                    # Verify actor was active at time of round announcement
                    # (had NDA, no prior drop in this cycle)
                    if _was_active_at(events, actor_id, ann_idx, cycle_ranges):
                        excluded = True
                        break
            if excluded:
                break

        if excluded:
            result[evt.event_id] = {
                "label": "DropTarget",
                "basis": (
                    f"Bidder in actor_ids was active but not in invited_actor_ids "
                    f"for a prior round announcement."
                ),
                "source_text": evt.drop_reason_text or "",
            }

    return result


def _was_active_at(
    events: list[SkillEventRecord],
    actor_id: str,
    target_idx: int,
    cycle_ranges: list[tuple[int, int]],
) -> bool:
    """Check if actor_id was active (had NDA, no drop) at target_idx within its cycle."""
    # Find which cycle target_idx belongs to
    for cycle_start, cycle_end in cycle_ranges:
        if cycle_start <= target_idx <= cycle_end:
            # Scan from cycle start to target_idx
            active = False
            for idx in range(cycle_start, target_idx):
                evt = events[idx]
                if actor_id in evt.actor_ids:
                    if evt.event_type == "nda":
                        active = True
                    elif evt.event_type == "drop":
                        active = False
            return active
    return False


# --- Contextual all_cash inference (D-08, D-09, D-10) ---

def _infer_all_cash_overrides(
    events: list[SkillEventRecord],
    cycles: list[dict],
) -> dict[str, bool]:
    """Cycle-local all_cash inference. Fail-closed per D-10.

    Only infers true when cash consideration is unambiguous:
    - Path 1: Executed event in cycle has explicit consideration_type="cash"
      -> propagate to untyped proposals in same cycle.
    - Path 2: No executed event, but ALL typed proposals are "cash"
      -> propagate to untyped proposals in same cycle.

    Never infers through mixed/CVR structures. Never overrides explicit
    non-null consideration_type on proposals.
    """
    result: dict[str, bool] = {}
    event_order = [e.event_id for e in events]

    for cycle in cycles:
        start_id = cycle["start_event_id"]
        end_id = cycle["end_event_id"]
        try:
            start_idx = event_order.index(start_id)
            end_idx = event_order.index(end_id)
        except ValueError:
            continue

        cycle_events = events[start_idx : end_idx + 1]

        # Find executed events in this cycle
        executed = [e for e in cycle_events if e.event_type == "executed"]

        # Collect proposals
        untyped_proposals = [
            e for e in cycle_events
            if e.event_type == "proposal"
            and (e.terms is None or e.terms.consideration_type is None)
        ]
        if not untyped_proposals:
            continue  # Nothing to infer

        if executed:
            # Path 1: Check executed event's consideration_type
            exec_terms = executed[0].terms
            if exec_terms and exec_terms.consideration_type == "cash":
                for evt in untyped_proposals:
                    result[evt.event_id] = True
                continue
            # Executed exists but is NOT explicit cash -> fail closed
            continue

        # Path 2: No executed event in cycle
        typed_proposals = [
            e for e in cycle_events
            if e.event_type == "proposal"
            and e.terms is not None
            and e.terms.consideration_type is not None
        ]
        if typed_proposals and all(
            e.terms.consideration_type == "cash" for e in typed_proposals
        ):
            for evt in untyped_proposals:
                result[evt.event_id] = True

    return result
```
