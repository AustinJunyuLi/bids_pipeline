    **Step 1: Create `skill_pipeline/complexity.py`.**
    ```python
    """Complexity-based deal classification for extraction routing."""

    from __future__ import annotations

    from typing import Literal

    from skill_pipeline.pipeline_models.source import ChronologyBlock

    SIMPLE_DEAL_MAX_BLOCKS: int = 150

    def classify_deal_complexity(
        blocks: list[ChronologyBlock],
        *,
        max_blocks: int = SIMPLE_DEAL_MAX_BLOCKS,
    ) -> Literal["simple", "complex"]:
        """Classify a deal as simple or complex based on block count.

        Simple deals use single-pass extraction (entire filing in one prompt).
        Complex deals use chunked extraction with overlap windows.

        Per D-07, the threshold defaults to 150 blocks. The 9-deal corpus splits
        6 simple / 3 complex at this boundary.
        """
        if len(blocks) <= max_blocks:
            return "simple"
        return "complex"
    ```
