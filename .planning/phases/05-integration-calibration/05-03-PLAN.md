    ```python
    def _summarize_db_load(paths) -> DbLoadStageSummary:
        db_path = paths.database_path
        if not db_path.exists():
            return DbLoadStageSummary(status=StageStatus.MISSING)
        try:
            from skill_pipeline.db_schema import open_pipeline_db
            con = open_pipeline_db(db_path, read_only=True)
            try:
                actor_rows = con.execute(
                    "SELECT COUNT(*) FROM actors WHERE deal_slug = ?",
                    [paths.deal_slug],
                ).fetchone()[0]
                event_rows = con.execute(
                    "SELECT COUNT(*) FROM events WHERE deal_slug = ?",
                    [paths.deal_slug],
                ).fetchone()[0]
                span_rows = con.execute(
                    "SELECT COUNT(*) FROM spans WHERE deal_slug = ?",
                    [paths.deal_slug],
                ).fetchone()[0]
            finally:
                con.close()
            if event_rows == 0:
                return DbLoadStageSummary(status=StageStatus.MISSING)
            return DbLoadStageSummary(
                status=StageStatus.PASS,
                actor_rows=actor_rows,
                event_rows=event_rows,
                span_rows=span_rows,
            )
        except Exception:
            return DbLoadStageSummary(status=StageStatus.FAIL)
    ```

    Add `_summarize_db_export(paths)`:
    ```python
    def _summarize_db_export(paths) -> DbExportStageSummary:
        csv_path = paths.deal_events_path
        if not csv_path.exists():
            return DbExportStageSummary(
                status=StageStatus.MISSING,
                output_path=csv_path,
            )
        # Count non-header event rows
        lines = csv_path.read_text(encoding="utf-8").strip().split("\n")
        # Header block is 2 lines, then event rows
        event_rows = max(0, len(lines) - 2) if len(lines) > 2 else 0
        status = StageStatus.PASS if event_rows > 0 else StageStatus.FAIL
        return DbExportStageSummary(
            status=status,
            event_rows=event_rows,
            output_path=csv_path,
        )
    ```

    ```python
    # Inline validation script (not a permanent test file):
    import csv, subprocess, sys
    from pathlib import Path

    seeds = list(csv.DictReader(open("data/seeds.csv")))
    results = []
    for row in seeds:
        slug = row["deal_slug"]
        # Check prerequisites
        base = Path(f"data/skill/{slug}")
        has_artifacts = all((base / "extract" / f).exists() for f in
            ["actors_raw.json", "events_raw.json", "spans.json"]) and \
            (base / "enrich" / "deterministic_enrichment.json").exists()

        if not has_artifacts:
            results.append((slug, "SKIP", "missing prerequisites"))
            continue

        rc_load = subprocess.run(["skill-pipeline", "db-load", "--deal", slug]).returncode
        rc_export = subprocess.run(["skill-pipeline", "db-export", "--deal", slug]).returncode
        status = "PASS" if rc_load == 0 and rc_export == 0 else "FAIL"
        results.append((slug, status, f"db-load={rc_load} db-export={rc_export}"))

    for slug, status, detail in results:

    passed = sum(1 for _, s, _ in results if s == "PASS")
    total = len(results)
    ```
